13  package biz.gabrys.maven.plugins.css.splitter.validation;
14  
15  import biz.gabrys.maven.plugins.css.splitter.css.type.NodeRule;
16  
17  abstract class AbstractRulePropertiesLimitValidator<T extends NodeRule> implements RulePropertiesLimitValidator {
18  
19      private final Class<T> clazz;
20  
21      protected AbstractRulePropertiesLimitValidator(final Class<T> clazz) {
22          this.clazz = clazz;
23      }
24  
25      @Override
26      public final boolean isSupportedType(final NodeRule rule) {
27          return rule != null && rule.getClass() == clazz;
28      }
29  
30      @Override
31      public final void validate(final NodeRule rule, final int limit) {
32          if (!isSupportedType(rule)) {
33              throw new IllegalArgumentException(String.format("Cannot cast an instance of the %s to the %s class!",
34                      rule == null ? "null" : rule.getClass().getName(), clazz.getName()));
35          }
36          validate2(clazz.cast(rule), limit);
37      }
38  
39      protected abstract void validate2(T rule, int limit);
40  
41      protected void throwException(final int current, final int limit, final String code) {
42          throw new ValidationException(
43                  String.format("The number of style properties (%d) in non-splittable rule exceeded the allowable limit (%d)!"
44                          + " CSS code that causes validation error:%n%s", current, limit, code));
45      }
46  }
