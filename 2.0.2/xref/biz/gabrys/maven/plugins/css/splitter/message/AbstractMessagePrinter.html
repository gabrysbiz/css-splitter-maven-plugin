13  package biz.gabrys.maven.plugins.css.splitter.message;
14  
15  import biz.gabrys.maven.plugins.css.splitter.css.type.NodeRule;
16  
17  abstract class AbstractMessagePrinter<T extends NodeRule> implements MessagePrinter {
18  
19      private final Class<T> clazz;
20  
21      protected AbstractMessagePrinter(final Class<T> clazz) {
22          this.clazz = clazz;
23      }
24  
25      @Override
26      public final boolean isSupportedType(final NodeRule rule) {
27          return rule != null && rule.getClass() == clazz;
28      }
29  
30      @Override
31      public final void print(final NodeRule rule) {
32          if (isSupportedType(rule)) {
33              print2(clazz.cast(rule));
34          }
35      }
36  
37      protected abstract void print2(T rule);
38  }
