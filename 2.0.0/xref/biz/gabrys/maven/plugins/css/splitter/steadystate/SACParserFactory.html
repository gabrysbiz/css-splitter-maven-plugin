13  package biz.gabrys.maven.plugins.css.splitter.steadystate;
14  
15  import java.lang.reflect.InvocationTargetException;
16  import java.util.EnumMap;
17  import java.util.Map;
18  
19  import com.steadystate.css.parser.SACParser;
20  import com.steadystate.css.parser.SACParserCSS1;
21  import com.steadystate.css.parser.SACParserCSS2;
22  import com.steadystate.css.parser.SACParserCSS21;
23  import com.steadystate.css.parser.SACParserCSS3;
24  
25  import biz.gabrys.maven.plugins.css.splitter.css.Standard;
26  
27  class SACParserFactory {
28  
29      private static final Map<Standard, Class<? extends SACParser>> PARSERS_BY_CSS_STANDARD;
30  
31      static {
32          PARSERS_BY_CSS_STANDARD = new EnumMap<Standard, Class<? extends SACParser>>(Standard.class);
33          PARSERS_BY_CSS_STANDARD.put(Standard.VERSION_1_0, SACParserCSS1.class);
34          PARSERS_BY_CSS_STANDARD.put(Standard.VERSION_2_0, SACParserCSS2.class);
35          PARSERS_BY_CSS_STANDARD.put(Standard.VERSION_2_1, SACParserCSS21.class);
36          PARSERS_BY_CSS_STANDARD.put(Standard.VERSION_3_0, SACParserCSS3.class);
37      }
38  
39      SACParser create(final Standard standard) {
40          try {
41              return PARSERS_BY_CSS_STANDARD.get(standard).getDeclaredConstructor().newInstance();
42          } catch (final InstantiationException e) {
43              throw new IllegalStateException(e);
44          } catch (final IllegalAccessException e) {
45              throw new IllegalStateException(e);
46          } catch (final IllegalArgumentException e) {
47              throw new IllegalStateException(e);
48          } catch (final InvocationTargetException e) {
49              throw new IllegalStateException(e);
50          } catch (final NoSuchMethodException e) {
51              throw new IllegalStateException(e);
52          } catch (final SecurityException e) {
53              throw new IllegalStateException(e);
54          }
55      }
56  }
