13  package biz.gabrys.maven.plugins.css.splitter.message;
14  
15  import org.apache.maven.plugin.logging.Log;
16  
17  import biz.gabrys.maven.plugins.css.splitter.css.type.UnknownRule;
18  
19  class UnknownRuleSizeMessagePrinter extends AbstractMessagePrinter<UnknownRule> {
20  
21      private final Log logger;
22  
23      UnknownRuleSizeMessagePrinter(final Log logger) {
24          super(UnknownRule.class);
25          this.logger = logger;
26      }
27  
28      @Override
29      public boolean isEnabled() {
30          return logger.isDebugEnabled();
31      }
32  
33      @Override
34      protected void print2(final UnknownRule rule) {
35          logger.debug("Found non-standard (unknown) rule:\n" + rule.getCode());
36          final int size = rule.getSize();
37          logger.debug(String.format("I treat that it contains %d propert%s.", size, size != 1 ? "ies" : "y"));
38      }
39  }
