13  package biz.gabrys.maven.plugins.css.splitter.split;
14  
15  import java.util.List;
16  
17  import biz.gabrys.maven.plugins.css.splitter.css.type.StyleProperty;
18  import biz.gabrys.maven.plugins.css.splitter.css.type.StyleRule;
19  
20  class StyleRuleSplitter extends AbstractRuleSplitter<StyleRule> {
21  
22      private final NeighborsManager neighborsManager;
23  
24      StyleRuleSplitter() {
25          this(new NeighborsManager());
26      }
27  
28      // fot tests
29      StyleRuleSplitter(final NeighborsManager neighborsManager) {
30          super(StyleRule.class);
31          this.neighborsManager = neighborsManager;
32      }
33  
34      @Override
35      protected boolean isSplittable2(final StyleRule rule) {
36          return rule.isSplittable();
37      }
38  
39      @Override
40      protected SplitResult split2(final StyleRule rule, final int splitAfter) {
41          final List<StyleProperty> properties = rule.getProperties();
42          final StyleRule first = new StyleRule(rule.getSelectors(), properties.subList(0, splitAfter));
43          final StyleRule second = new StyleRule(rule.getSelectors(), properties.subList(splitAfter, properties.size()));
44          neighborsManager.fill(rule, first, second);
45          return new SplitResult(first, second);
46      }
47  }
