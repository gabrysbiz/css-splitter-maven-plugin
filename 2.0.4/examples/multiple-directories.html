Multiple source/output directories

If you want to use more than one source or output directory, you must configure multiple executions. 

<project>
   ...
   <build>
      <plugins>
         ...
         <plugin>
            <groupId>biz.gabrys.maven.plugins</groupId>
            <artifactId>css-splitter-maven-plugin</artifactId>
            <version>2.0.4-SNAPSHOT</version>
            <executions>
               <execution>
                  <id>split-common</id>
                  <goals>
                     <goal>split</goal>
                  </goals>
                  <configuration>
                     <sourceDirectory>src/main/css/common</sourceDirectory>
                     <outputDirectory>target/css/common</outputDirectory>
                  </configuration>
               </execution>
               <execution>
                  <id>split-vendors</id>
                  <goals>
                     <goal>split</goal>
                  </goals>
                  <configuration>
                     <sourceDirectory>src/vendors</sourceDirectory>
                     <outputDirectory>target/css/vendors</outputDirectory>
                  </configuration>
               </execution>
               ...
            </executions>
         </plugin>
      </plugins>
   </build>
