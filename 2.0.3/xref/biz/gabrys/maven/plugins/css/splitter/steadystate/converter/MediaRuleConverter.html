13  package biz.gabrys.maven.plugins.css.splitter.steadystate.converter;
14  
15  import java.util.ArrayList;
16  import java.util.LinkedList;
17  import java.util.List;
18  
19  import org.w3c.dom.css.CSSRuleList;
20  
21  import com.steadystate.css.dom.CSSMediaRuleImpl;
22  import com.steadystate.css.dom.MediaListImpl;
23  
24  import biz.gabrys.maven.plugins.css.splitter.css.Standard;
25  import biz.gabrys.maven.plugins.css.splitter.css.type.ComplexRule;
26  import biz.gabrys.maven.plugins.css.splitter.css.type.NodeRule;
27  
28  class MediaRuleConverter extends AbstractRuleConverter<CSSMediaRuleImpl, ComplexRule> {
29  
30      private final RuleConverter<?> converter;
31      private final CssFormatter formatter;
32  
33      MediaRuleConverter(final Standard standard, final boolean strict) {
34          super(CSSMediaRuleImpl.class);
35          converter = createConverter(this, standard, strict);
36          formatter = new CssFormatter();
37      }
38  
39      // for tests
40      MediaRuleConverter(final RuleConverter<?> converter, final CssFormatter formatter) {
41          super(CSSMediaRuleImpl.class);
42          this.converter = converter;
43          this.formatter = formatter;
44      }
45  
46      static RuleConverter<?> createConverter(final MediaRuleConverter thisObject, final Standard standard, final boolean strict) {
47          if (strict && Standard.VERSION_3_0 != standard) {
48              return new StyleRuleConverter();
49          }
50  
51          final List<RuleConverter<?>> converters = new ArrayList<RuleConverter<?>>();
52          converters.add(new StyleRuleConverter());
53          if (Standard.VERSION_3_0 == standard) {
54              converters.add(thisObject);
55          }
56          if (!strict) {
57              converters.add(new PageRuleConverter());
58              converters.add(new UnknownRuleConverter());
59          }
60          return new MultipleRuleConverter(converters);
61      }
62  
63      @Override
64      protected ComplexRule convert2(final CSSMediaRuleImpl rule) {
65          final List<String> selectors = new LinkedList<String>();
66          final MediaListImpl mediaList = (MediaListImpl) rule.getMedia();
67          for (int i = 0; i < mediaList.getLength(); ++i) {
68              selectors.add(formatter.format(mediaList.mediaQuery(i)));
69          }
70  
71          final List<NodeRule> rules = new LinkedList<NodeRule>();
72          final CSSRuleList ruleList = rule.getCssRules();
73          for (int i = 0; i < ruleList.getLength(); ++i) {
74              rules.add(converter.convert(ruleList.item(i)));
75          }
76  
77          return new ComplexRule("@media", selectors, rules);
78      }
79  }
