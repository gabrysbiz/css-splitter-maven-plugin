13  package biz.gabrys.maven.plugins.css.splitter.compressor;
14  
15  import java.io.IOException;
16  import java.io.StringReader;
17  import java.io.StringWriter;
18  
19  import com.yahoo.platform.yui.compressor.CssCompressor;
20  
21  public final class CodeCompressor {
22  
23      private final int lineBreak;
24  
25      public CodeCompressor(final int lineBreak) {
26          this.lineBreak = lineBreak;
27      }
28  
29      public String compress(final String css) {
30          CssCompressor compressor;
31          try {
32              compressor = new CssCompressor(new StringReader(css));
33          } catch (final IOException e) {
34              throw new InitializationException(e);
35          }
36          final StringWriter compressed = new StringWriter();
37          try {
38              compressor.compress(compressed, lineBreak);
39          } catch (final IOException e) {
40              throw new CompressionException(e);
41          }
42          return compressed.toString();
43      }
44  }
