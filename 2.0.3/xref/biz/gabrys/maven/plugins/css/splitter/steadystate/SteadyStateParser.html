13  package biz.gabrys.maven.plugins.css.splitter.steadystate;
14  
15  import java.io.IOException;
16  import java.io.StringReader;
17  
18  import org.apache.maven.plugin.logging.Log;
19  import org.w3c.css.sac.InputSource;
20  import org.w3c.dom.css.CSSStyleSheet;
21  
22  import com.steadystate.css.parser.CSSOMParser;
23  import com.steadystate.css.parser.SACParser;
24  
25  import biz.gabrys.maven.plugins.css.splitter.css.type.StyleSheet;
26  import biz.gabrys.maven.plugins.css.splitter.steadystate.converter.StyleSheetConverter;
27  
28  public class SteadyStateParser {
29  
30      private final Log logger;
31  
32      public SteadyStateParser(final Log logger) {
33          this.logger = logger;
34      }
35  
36      public StyleSheet parse(final String css, final ParserOptions options) {
37          final SACParserFactory factory = new SACParserFactory();
38          final SACParser sacParser = factory.create(options.getStandard());
39          sacParser.setIeStarHackAccepted(options.isStarHackAllowed());
40  
41          final CSSOMParser parser = new CSSOMParser(sacParser);
42  
43          final SteadyErrorHandler errorHandler = new SteadyErrorHandler(logger);
44          parser.setErrorHandler(errorHandler);
45  
46          final InputSource source = new InputSource(new StringReader(css));
47          CSSStyleSheet stylesheet;
48          try {
49              stylesheet = parser.parseStyleSheet(source, null, null);
50          } catch (final IOException e) {
51              throw new ParserException(e);
52          }
53          errorHandler.validate();
54  
55          final StyleSheetConverter converter = new StyleSheetConverter(options.getStandard(), options.isStrict());
56          return converter.convert(stylesheet);
57      }
58  }
