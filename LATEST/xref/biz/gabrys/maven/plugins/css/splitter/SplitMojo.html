353     private String getDefaultCacheTokenValue() {
354         if (TokenType.DATE.name().equalsIgnoreCase(cacheTokenType)) {
355             return "yyyyMMddHHmmss";
356         } else {
357             return null;
358         }

361     private void calculateParameters() {
362         if (getLog().isDebugEnabled()) {
363             verbose = true;
364         }
365         if (includes.length == 0) {
366             includes = getDefaultIncludes();
367         }
368         if (maxRules < 1) {
369             maxRules = Integer.MAX_VALUE;
370         }
371         if (rulesLimit < 1) {
372             rulesLimit = Integer.MAX_VALUE;
373         }
374         if (maxImports < OrderedTree.MIN_NUMBER_OF_CHILDREN) {
375             maxImports = Integer.MAX_VALUE;
376         }
377         if (importsDepthLimit < 1) {
378             importsDepthLimit = Integer.MAX_VALUE;
379         }
380         if (cacheTokenValue == null) {
381             cacheTokenValue = getDefaultCacheTokenValue();
382         }
383     }

391     public void execute() throws MojoExecutionException, MojoFailureException {
392         logParameters();
393         if (skip) {
394             getLog().info("Skipping job execution");
395             return;
396         }
397         calculateParameters();
398         validateParameters();
399         runSplitter();
400     }

402     private void runSplitter() throws MojoFailureException {
403         if (!sourceDirectory.exists()) {
404             getLog().warn("Source directory does not exist: " + sourceDirectory.getAbsolutePath());
405             return;
406         }
407         final Collection<File> files = getFiles();
408         if (files.isEmpty()) {
409             getLog().warn("No sources to split");
410             return;
411         }
412         resolveCacheToken();
413         splitFiles(files);
414     }
415 
416     private Collection<File> getFiles() {
417         final ScannerPatternFormat patternFormat = ScannerPatternFormat.toPattern(filesetPatternFormat);
418         final FileScanner scanner = new ScannerFactory().create(patternFormat, getLog());
419         if (verbose) {
420             getLog().info("Scanning directory for sources...");
421         }
422         return scanner.getFiles(sourceDirectory, includes, excludes);
423     }

425     private void resolveCacheToken() {
426         if (getLog().isDebugEnabled()) {
427             getLog().debug("Resolving cache token...");
428         }
429 
430         if (!TokenType.NONE.name().equalsIgnoreCase(cacheTokenType)) {
431             final String value = TokenType.create(cacheTokenType).createFactory().create(cacheTokenValue);
432             final StringBuilder cacheToken = new StringBuilder();
433             cacheToken.append(UrlEscaper.escape(cacheTokenParameter));
434             cacheToken.append('=');
435             cacheToken.append(UrlEscaper.escape(value));
436             resolvedCacheToken = cacheToken.toString();
437         }
438 
439         if (verbose) {
440             getLog().info("Cache token: " + resolvedCacheToken);
441         }
442     }

444     private void splitFiles(final Collection<File> sources) throws MojoFailureException {
445         final String sourceFilesText = "source file" + (sources.size() != 1 ? "s" : "");
446         getLog().info("Splitting " + sources.size() + ' ' + sourceFilesText + " to " + outputDirectory.getAbsolutePath());
447         final Timer timer = SystemTimer.getStartedTimer();
448         for (final File source : sources) {
449             if (isCompilationRequired(source)) {
450                 splitFile(source);
451             } else if (verbose) {
452                 getLog().info("Skipping stylesheet split (not modified): " + source.getAbsolutePath());
453             }
454         }
455         getLog().info("Finished " + sourceFilesText + " split in " + timer.stop());
456     }
457 
458     private boolean isCompilationRequired(final File source) {
459         if (force) {
460             return true;
461         }
462         final File destination = new DestinationFileCreator(sourceDirectory, outputDirectory, outputFileNamePattern).create(source);
463         if (!destination.exists()) {
464             return true;
465         }
466         return source.lastModified() > destination.lastModified();
467     }
468 
469     private void splitFile(final File source) throws MojoFailureException {
470         Timer timer = null;
471         if (verbose) {
472             getLog().info("Splitting stylesheet: " + source.getAbsolutePath());
473             timer = SystemTimer.getStartedTimer();
474         }
475         final String css = readCss(source);
476         final List<StyleSheet> parts;
477         try {
478             if (getLog().isDebugEnabled()) {
479                 getLog().debug("Parsing stylesheet...");
480             }
481             final StyleSheet stylesheet = new SteadyStateParser(getLog()).parse(css, Standard.create(standard));
482             validateStyleSheet(stylesheet);
483             if (getLog().isDebugEnabled()) {
484                 getLog().debug("Splitting stylesheet to parts...");
485             }
486             parts = new Splliter(maxRules).split(stylesheet);
487         } catch (final Exception e) {
488             throw new MojoFailureException(e.getMessage(), e);
489         }
490         if (verbose) {
491             getLog().info(String.format("Split to %d stylesheet%s.", parts.size(), parts.size() == 1 ? "" : "s"));
492         }
493         saveParts(source, parts);
494         if (timer != null) {
495             getLog().info("Finished in " + timer.stop());
496         }
497     }

499     private String readCss(final File file) throws MojoFailureException {
500         try {
501             return FileUtils.readFileToString(file, encoding);
502         } catch (final IOException e) {
503             throw new MojoFailureException("Cannot read file: " + file.getAbsolutePath(), e);
504         }
505     }
506 
507     private void validateStyleSheet(final StyleSheet stylesheet) throws ValidationException {
508         if (getLog().isDebugEnabled()) {
509             getLog().debug("Validating stylesheet...");
510         }
511 
512         final Collection<StyleSheetValidator> validators = new ArrayList<StyleSheetValidator>();
513         StyleSheetCounter counter = new StyleSheetCounterImpl();
514         if (verbose) {
515             counter = new LoggingStyleSheetCounter(counter, getLog());
516         }
517         validators.add(new RulesLimitValidator(rulesLimit, counter));
518         validators.add(new StylePropertiesLimitValidator(maxRules, getLog()));
519 
520         for (final StyleSheetValidator validator : validators) {
521             validator.validate(stylesheet);
522         }
523     }
524 
525     private void saveParts(final File source, final List<StyleSheet> parts) throws MojoFailureException {
526         if (getLog().isDebugEnabled()) {
527             getLog().debug("Creating imports tree...");
528         }
529         final OrderedTree<StyleSheet> stylesheetsTree = new OrderedTree<StyleSheet>(parts, maxImports);
530         validateImportsDepth(stylesheetsTree);
531 
532         if (verbose) {
533             getLog().info("Saving CSS code...");
534         }
535         final int numberOfDigits = String.valueOf(stylesheetsTree.size()).length();
536         final String indexPattern = "%0" + numberOfDigits + 'd';
537         saveStyleSheetsTree(source, stylesheetsTree, indexPattern);
538     }
539 
540     private void validateImportsDepth(final OrderedTreeNode<StyleSheet> tree) throws MojoFailureException {
541         final int depth = tree.getDepth();
542         if (verbose) {
543             getLog().info("Imports depth: " + depth);
544         }
545         if (getLog().isDebugEnabled()) {
546             getLog().debug("Validating imports depth...");
547         }
548         if (depth > importsDepthLimit) {
549             throw new MojoFailureException(
550                     String.format("The number of @import depth (%d) exceeded the allowable limit (%d)!", depth, importsDepthLimit));
551         }
552     }

554     private void saveStyleSheetsTree(final File source, final OrderedTreeNode<StyleSheet> node, final String indexPattern)
555             throws MojoFailureException {
556         final DestinationFileCreator fileCreator = new DestinationFileCreator(sourceDirectory, outputDirectory);
557         if (node.getOrder() == 0) {
558             fileCreator.setFileNamePattern(outputFileNamePattern);
559         } else {
560             final String index = String.format(indexPattern, node.getOrder());
561             fileCreator.setFileNamePattern(outputPartNamePattern.replace(PART_INDEX_PARAMETER, index));
562         }
563         final File target = fileCreator.create(source);
564 
565         if (node.hasValue()) {
566             saveCss(target, node.getValue().toString());
567             return;
568         }
569 
570         final StringBuilder imports = new StringBuilder();
571         for (final OrderedTreeNode<StyleSheet> child : node.getChildren()) {
572             final String index = String.format(indexPattern, child.getOrder());
573             fileCreator.setFileNamePattern(outputPartNamePattern.replace(PART_INDEX_PARAMETER, index));
574             final File childTarget = fileCreator.create(source);
575             saveStyleSheetsTree(source, child, indexPattern);
576             imports.append(String.format("@import \"%s?%s\";%n", childTarget.getName(), resolvedCacheToken));
577         }
578         saveCss(target, imports.toString());
579     }

581     private void saveCss(final File file, final String css) throws MojoFailureException {
582         if (getLog().isDebugEnabled()) {
583             getLog().debug("Saving CSS code to " + file.getAbsolutePath());
584         }
585         try {
586             FileUtils.write(file, css, encoding);
587         } catch (final IOException e) {
588             throw new MojoFailureException("Cannot save file: " + file.getAbsolutePath(), e);
589         }
590     }
