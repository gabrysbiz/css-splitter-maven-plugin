13  package biz.gabrys.maven.plugins.css.splitter.steadystate;
14  
15  import java.io.IOException;
16  import java.io.StringReader;
17  
18  import org.apache.maven.plugin.logging.Log;
19  import org.w3c.css.sac.InputSource;
20  import org.w3c.dom.css.CSSStyleSheet;
21  
22  import com.steadystate.css.parser.CSSOMParser;
23  
24  import biz.gabrys.maven.plugins.css.splitter.css.types.StyleSheet;
25  import biz.gabrys.maven.plugins.css.splitter.steadystate.converters.StyleSheetConverter;
26  
27  public class SteadyStateParser {
28  
29      private final Log logger;
30  
31      public SteadyStateParser(final Log logger) {
32          this.logger = logger;
33      }
34  
35      public StyleSheet parse(final String css, final ParserOptions options) {
36          final NativeParserFactory factory = new NativeParserFactory();
37          final CSSOMParser parser = new CSSOMParser(factory.create(options.getStandard()));
38  
39          final SteadyErrorHandler errorHandler = new SteadyErrorHandler(logger);
40          parser.setErrorHandler(errorHandler);
41  
42          final InputSource source = new InputSource(new StringReader(css));
43          CSSStyleSheet stylesheet;
44          try {
45              stylesheet = parser.parseStyleSheet(source, null, null);
46          } catch (final IOException e) {
47              throw new ParserException(e);
48          }
49          errorHandler.validate();
50  
51          final StyleSheetConverter converter = new StyleSheetConverter(options.isStrict());
52          return converter.convert(stylesheet);
53      }
54  }
