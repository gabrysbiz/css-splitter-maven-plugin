13  package biz.gabrys.maven.plugins.css.splitter.steadystate.converter;
14  
15  import java.util.Arrays;
16  import java.util.LinkedList;
17  import java.util.List;
18  
19  import com.steadystate.css.dom.CSSFontFaceRuleImpl;
20  import com.steadystate.css.dom.CSSStyleDeclarationImpl;
21  import com.steadystate.css.dom.Property;
22  
23  import biz.gabrys.maven.plugins.css.splitter.css.type.StyleProperty;
24  import biz.gabrys.maven.plugins.css.splitter.css.type.StyleRule;
25  
26  class FontFaceRuleConverter extends AbstractRuleConverter<CSSFontFaceRuleImpl, StyleRule> {
27  
28      private final StylePropertyConverter converter;
29  
30      FontFaceRuleConverter() {
31          this(new StylePropertyConverter());
32      }
33  
34      // for tests
35      FontFaceRuleConverter(final StylePropertyConverter converter) {
36          super(CSSFontFaceRuleImpl.class);
37          this.converter = converter;
38      }
39  
40      @Override
41      protected StyleRule convert2(final CSSFontFaceRuleImpl rule) {
42          final CSSStyleDeclarationImpl style = (CSSStyleDeclarationImpl) rule.getStyle();
43          final List<StyleProperty> properties = new LinkedList<StyleProperty>();
44          for (final Property property : style.getProperties()) {
45              properties.add(converter.convert(property));
46          }
47          return new StyleRule(Arrays.asList("@font-face"), properties, false);
48      }
49  }
