13  package biz.gabrys.maven.plugins.css.splitter.validation;
14  
15  import biz.gabrys.maven.plugins.css.splitter.css.type.StyleRule;
16  
17  final class StyleRulePropertiesLimitValidator extends AbstractRulePropertiesLimitValidator<StyleRule> {
18  
19      StyleRulePropertiesLimitValidator() {
20          super(StyleRule.class);
21      }
22  
23      @Override
24      protected void validate2(final StyleRule rule, final int limit) {
25          if (rule.isSplittable()) {
26              return;
27          }
28          final int size = rule.getSize();
29          if (size > limit) {
30              throwException(size, limit, rule.getCode());
31          }
32      }
33  }
