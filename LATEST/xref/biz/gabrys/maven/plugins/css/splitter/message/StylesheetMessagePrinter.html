13  package biz.gabrys.maven.plugins.css.splitter.message;
14  
15  import org.apache.maven.plugin.logging.Log;
16  
17  import biz.gabrys.maven.plugins.css.splitter.css.type.NodeRule;
18  import biz.gabrys.maven.plugins.css.splitter.css.type.StyleSheet;
19  
20  public class StylesheetMessagePrinter {
21  
22      private final MessagePrinter printer;
23  
24      public StylesheetMessagePrinter(final Log logger, final boolean strict) {
25          printer = new TreeMessagePrinter(new AnyMessagePrinter(logger, strict));
26      }
27  
28      // for tests
29      StylesheetMessagePrinter(final MessagePrinter printer) {
30          this.printer = printer;
31      }
32  
33      public void print(final StyleSheet stylesheet) {
34          if (printer.isEnabled()) {
35              for (final NodeRule rule : stylesheet.getRules()) {
36                  printer.print(rule);
37              }
38          }
39      }
40  }
