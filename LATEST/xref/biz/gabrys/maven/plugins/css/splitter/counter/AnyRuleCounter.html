13  package biz.gabrys.maven.plugins.css.splitter.counter;
14  
15  import java.util.ArrayList;
16  import java.util.Arrays;
17  import java.util.List;
18  
19  import biz.gabrys.maven.plugins.css.splitter.css.types.NodeRule;
20  
21  public class AnyRuleCounter implements RuleCounter {
22  
23      private final List<RuleCounter> counters;
24  
25      public AnyRuleCounter() {
26          this(Arrays.<RuleCounter>asList(new ComplexRuleCounter(), new StyleRuleCounter(), new UnknownRuleCounter()));
27      }
28  
29      // for tests
30      AnyRuleCounter(final List<RuleCounter> counters) {
31          this.counters = new ArrayList<RuleCounter>(counters);
32      }
33  
34      public boolean isSupportedType(final NodeRule rule) {
35          return true;
36      }
37  
38      public int count(final NodeRule rule) {
39          for (final RuleCounter counter : counters) {
40              if (counter.isSupportedType(rule)) {
41                  return counter.count(rule);
42              }
43          }
44          return 0;
45      }
46  }
